#Builds the docker image
docker build -t ltt_backpack:ltt_backpack /mnt/user/Documents/ltt_backpack

#Removes the docker image
docker rmi -f 7ae7fb8b6ec4

#Runs the image in a docker container, removes the log file, and removes the 
#container (Unraid)
#!/bin/bash
: > $(docker inspect --format='{{.LogPath}}' 7ae7fb8b6ec4)
docker run --name ltt_backpack -v /mnt/user/Documents/ltt_backpack:/app 7ae7fb8b6ec4
docker rm ltt_backpack
